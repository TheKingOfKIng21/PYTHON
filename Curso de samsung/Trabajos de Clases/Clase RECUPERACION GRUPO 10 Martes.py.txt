for i in range (1, numero+1, 2): #1->5     1,3,5
    for j in range(i, 0, -2):  #1->0 1, -1    3 1  5 3 1

# Escribir un programa que pida al usuario un número entero positivo mayor que 2 y muestre por pantalla si es un número primo o no.

# In[42]:


n = int(input("Introduce un numero entero positivo mayor que 2: "))

i=2

while n%i != 0:
    i+=1    
    if i==n:
        print(str(n) + " es primo")
    else:
        print(str(n) + " no es primo")
