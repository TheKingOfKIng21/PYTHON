Construir una función que construya un DataFrame a partir del un fichero con el formato anterior y devuelva otro DataFrame con el mínimo, el máximo y la media de dada columna.

# In[28]:


import pandas as pd

def resumen_cotizaciones(fichero): #Recibe el nombre del archivo que se va a leer(Cargar)
    Datos = pd.read_csv(fichero, sep=';', decimal=',', thousands='.', index_col=0) #Con esta linea estamos cargando el fichero
    return pd.DataFrame([Datos.min(), Datos.max(), Datos.mean()], index=['Minimo', 'Maximo', 'Media'])

resumen_cotizaciones('cotizacionIbex.csv')
